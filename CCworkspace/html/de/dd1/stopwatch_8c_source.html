83 // @fn update_stopwatch_timer
84 // @brief Set new compare time for next 1/1Hz or 1/100Hz interrupt. Takes care for exact 1 second timing.
85 // @param ticks (1 tick = 1/32768 sec)
86 // @return none
87 // *************************************************************************************************
88 void update_stopwatch_timer(void)
89 {
90  u16 value;
91 
92  // Load CCR register with next capture time
93  if (sStopwatch.viewStyle == DISPLAY_DEFAULT_VIEW)
94  {
95  // Timer interrupts occur every 32768/100 = 328 ACLK
96  // --> stopwatch runs too slow (1 sec nominal != 100 interupts * 328 ACLK = 32800 ACLK = 1.00098 sec)
97  // --> ideally correct timer value every 10 ticks by (32768 - 32800)/10 = 3.2
98  // --> correct timer value every 10Hz by 3,
99  // --> correct timer value every 1Hz correct by 5
100  value = TA0CCR2 + STOPWATCH_100HZ_TICK;
101 
102  if (sStopwatch.swtIs1Hz)
103  {
104  value -= 5;
105  sStopwatch.swtIs1Hz = 0;
106  sStopwatch.swtIs10Hz = 0;
107  }
108  else if (sStopwatch.swtIs10Hz)
109  {
110  value -= 3;
111  sStopwatch.swtIs10Hz = 0;
112  }
113  }
114  else // Alternative view
115  {
116  // Timer interrupts occur every 32768/1 = 32768 ACLK
117  value = TA0CCR2 + STOPWATCH_1HZ_TICK;
118  }
119 
120  // Update CCR
121  TA0CCR2 = value;
122 }

125 // *************************************************************************************************
126 // @fn stopwatch_tick
127 // @brief Called by 1/100Hz interrupt handler.
128 // Increases stopwatch counter and triggers display update.
129 // @param none
130 // @return none
131 // *************************************************************************************************
132 void stopwatch_tick(void)
133 {
134  static u8 delay = 0;
135 
136  // Default view (< 20 minutes): display and count MM:SS:hh
137  if (sStopwatch.viewStyle == DISPLAY_DEFAULT_VIEW)
138  {
139  // Add 1/100 sec
140  sStopwatch.time[7]++;
141 
142  // Draw flag minimizes display update activity
143  //
144  // swt.drawFlag = 1: second L
145  // swt.drawFlag = 2: second H/L
146  // swt.drawFlag = 3: minutes L, second H/L
147  // swt.drawFlag = 4: minutes H/L, second H/L
148  // swt.drawFlag = 5: hours L, minutes H/L, second H/L
149  // swt.drawFlag = 6: hours H/L, minutes H/L, second H/L
150  // swt.drawFlag = 7: 1/10 sec, 1/100 sec
151  // swt.drawFlag = 8: 1/100 sec (every 17/100 sec to reduce display draw activity)
152  if (delay++ > 17)
153  {
154  sStopwatch.drawFlag = 8;
155  delay = 0;
156  }
157 
158  // Add 1/10 sec
159  if (sStopwatch.time[7] == 0x3A)
160  {
161  sStopwatch.time[7]='0';
162  sStopwatch.time[6]++;
163 
164  // 1/10Hz trigger
165  sStopwatch.swtIs10Hz = 1;
166 
167  // Update draw flag
168  sStopwatch.drawFlag = 7;
169  }
170  }
171  else // Alternative view (20 minutes .. 20 hours): display and count HH:MM:SS
172  {
173  // Just add 1 second
174  sStopwatch.time[6] = 0x3A;
175  }
176 
177  // Second overflow?
178  if (sStopwatch.time[6] == 0x3A)
179  {
180  // Reset draw flag
181  sStopwatch.drawFlag = 1;
182 
183  // 1Hz trigger
184  sStopwatch.swtIs1Hz = 1;
185 
186  // Add data
187  sStopwatch.time[6]='0';
188  sStopwatch.time[5]++; // second L (0 - 9)
189  if (sStopwatch.time[5] == 0x3A)
190  {
191  sStopwatch.drawFlag++; // 2
192  sStopwatch.time[5] = '0';
193  sStopwatch.time[4]++; // second H (0 - 5)
194  if (sStopwatch.time[4] == '6')
195  {
196  sStopwatch.drawFlag ++; // 3
197  sStopwatch.time[4] = '0';
198  sStopwatch.time[3]++; // minutes L (0 - 9)
199  if (sStopwatch.time[3] == 0x3A)
200  {
201  sStopwatch.drawFlag++; // 4
202  sStopwatch.time[3] = '0';
203  sStopwatch.time[2]++; // minutes H (0 - 5)
204  if (sStopwatch.time[2] == '2')
205  {
206  // SWT display changes from MM:SS:hh to HH:MM:SS when reaching 20 minutes
207  sStopwatch.viewStyle = DISPLAY_ALTERNATIVE_VIEW;
208  display_stopwatch(LINE2, DISPLAY_LINE_UPDATE_FULL);
209 
210  }
211  else if (sStopwatch.time[2] == '6')
212  {
213  sStopwatch.drawFlag++; // 5
214  sStopwatch.time[2] = '0';
215  sStopwatch.time[1]++; // hours L (0-9)
216 
217  if (sStopwatch.time[1] == 0x3A)
218  {
219  sStopwatch.drawFlag++; // 6
220  sStopwatch.time[1] = '0';
221  sStopwatch.time[0]++; // hours H (0-1)
222 
223  if (sStopwatch.time[0] == '2')
224  {
225  // When reaching 20 hours, start over
226  reset_stopwatch();
227  sStopwatch.state = STOPWATCH_RUN;
228  display_stopwatch(LINE2, DISPLAY_LINE_UPDATE_FULL);
229  }
230  }
231  }
232  }
233  }
234  }
235  }
236 
237  // Always set display update flag
238  display.flag.update_stopwatch = 1;
239 }

244 // @fn reset_stopwatch
245 // @brief Clears stopwatch counter and sets stopwatch state to off.
246 // @param none
247 // @return none
248 // *************************************************************************************************
249 void reset_stopwatch(void)
250 {
251  // Clear counter
252  memcpy(sStopwatch.time, "00000000", sizeof(sStopwatch.time));
253 
254  // Clear trigger
255  sStopwatch.swtIs10Hz = 0; // 1/10Hz trigger
256  sStopwatch.swtIs1Hz = 0; // 1Hz trigger
257 
258  // Init stopwatch state 'Off'
259  sStopwatch.state = STOPWATCH_STOP;
260 
261  // Default display style is MM:SS:HH
262  sStopwatch.viewStyle = DISPLAY_DEFAULT_VIEW;
263 }

278 // *************************************************************************************************
279 // @fn start_stopwatch
280 // @brief Starts stopwatch timer interrupt and sets stopwatch state to on.
281 // @param none
282 // @return none
283 // *************************************************************************************************
284 void start_stopwatch(void)
285 {
286  // Set stopwatch run flag
287  sStopwatch.state = STOPWATCH_RUN;
288 
289  // Init CCR register with current time
290  TA0CCR2 = TA0R;
291 
292  // Load CCR register with next capture time
293  update_stopwatch_timer();
294 
295  // Reset IRQ flag
296  TA0CCTL2 &= ~CCIFG;
297 
298  // Enable timer interrupt
299  TA0CCTL2 |= CCIE;
300 
301  // Set stopwatch icon
302  display_symbol(LCD_ICON_STOPWATCH, SEG_ON);
303 }

306 // *************************************************************************************************
307 // @fn stop_stopwatch
308 // @brief Stops stopwatch timer interrupt and sets stopwatch state to off.
309 // Does not reset stopwatch count.
310 // @param none
311 // @return none
312 // *************************************************************************************************
313 void stop_stopwatch(void)
314 {
315  // Clear timer interrupt enable
316  TA0CCTL2 &= ~CCIE;
317 
318  // Clear stopwatch run flag
319  sStopwatch.state = STOPWATCH_STOP;
320 
321  // Clear stopwatch icon
322  display_symbol(LCD_ICON_STOPWATCH, SEG_OFF);
323 
324  // Call draw routine immediately
325  display_stopwatch(LINE2, DISPLAY_LINE_UPDATE_FULL);
326 }

374 // *************************************************************************************************
375 // @fn display_stopwatch
376 // @brief Stopwatch user routine. Sx starts/stops stopwatch, but does not reset count.
377 // @param u8 line LINE2
378 // u8 update DISPLAY_LINE_UPDATE_PARTIAL, DISPLAY_LINE_UPDATE_FULL
379 // @return none
380 // *************************************************************************************************
381 void display_stopwatch(u8 line, u8 update)
382 {
383  // Partial line update only
384  if (update == DISPLAY_LINE_UPDATE_PARTIAL)
385  {
386  if (display.flag.update_stopwatch)
387  {
388  if (sStopwatch.viewStyle == DISPLAY_DEFAULT_VIEW)
389  {
390  // Display MM:SS:hh
391 
392  // Check draw flag to minimize workload
393  if(sStopwatch.drawFlag != 0)
394  {
395  switch(sStopwatch.drawFlag)
396  {
397  case 4: display_char(LCD_SEG_L2_5, sStopwatch.time[2], SEG_ON);
398  case 3: display_char(LCD_SEG_L2_4, sStopwatch.time[3], SEG_ON);
399  case 2: display_char(LCD_SEG_L2_3, sStopwatch.time[4], SEG_ON);
400  case 1: display_char(LCD_SEG_L2_2, sStopwatch.time[5], SEG_ON);
401  case 7: display_char(LCD_SEG_L2_1, sStopwatch.time[6], SEG_ON);
402  case 8: display_char(LCD_SEG_L2_0, sStopwatch.time[7], SEG_ON);
403  }
404  }
405  }
406  else // DISPLAY_ALTERNATIVE_VIEW
407  {
408  // Display HH:MM:SS
409  switch(sStopwatch.drawFlag)
410  {
411  case 6: display_char(LCD_SEG_L2_5, sStopwatch.time[0], SEG_ON);
412  case 5: display_char(LCD_SEG_L2_4, sStopwatch.time[1], SEG_ON);
413  case 4: display_char(LCD_SEG_L2_3, sStopwatch.time[2], SEG_ON);
414  case 3: display_char(LCD_SEG_L2_2, sStopwatch.time[3], SEG_ON);
415  case 2: display_char(LCD_SEG_L2_1, sStopwatch.time[4], SEG_ON);
416  case 1: display_char(LCD_SEG_L2_0, sStopwatch.time[5], SEG_ON);
417  }
418  }
419  }
420  }
421  // Redraw whole line
422  else if (update == DISPLAY_LINE_UPDATE_FULL)
423  {
424  if (sStopwatch.viewStyle == DISPLAY_DEFAULT_VIEW)
425  {
426  // Display MM:SS:hh
427  display_chars(LCD_SEG_L2_5_0, sStopwatch.time+2, SEG_ON);
428  }
429  else // DISPLAY_ALTERNATIVE_VIEW
430  {
431  // Display HH:MM:SS
432  display_chars(LCD_SEG_L2_5_0, sStopwatch.time, SEG_ON);
433  }
434  display_symbol(LCD_SEG_L2_COL1, SEG_ON);
435  display_symbol(LCD_SEG_L2_COL0, SEG_ON);
436  }
437  else if (update == DISPLAY_LINE_CLEAR)
438  {
439  // Clean up symbols when leaving function
440  }
441 }
