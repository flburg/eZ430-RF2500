32 /* =-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=
33  * BSP (Board Support Package)
34  * Target : Texas Instruments EZ430-RF2500
35  * "MSP430 Wireless Development Tool"
36  * Board configuration file.
37  * =-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=
38  */
39 
40 #ifndef BSP_CONFIG_H
41 #define BSP_CONFIG_H
42 
43 
44 /**************************************************************************************************
45  * Configuration *
46  **************************************************************************************************
47  */
48 
49 /*
50  * Supported clock speeds : 1, 2, 4, 6, 8, 10, 12, and 16 MHz.
51  * The clock configuration values for 1, 8, 12, 16 MHz are taken from configuration flash
52  * set at the factory and should be the most accurate.
53  *
54  * NOTE! The clock speeds are approximate as they are derived from an internal
55  * digitally controlled oscillator.
56  */
57 #define BSP_CONFIG_CLOCK_MHZ_SELECT 8 /* approximate MHz */
58 
59 /*
60  * Custom clock configuration is available. Provide values for BCSCTL1 and DCOCTL as
61  * shown in the example below. The clock speed itself must be provided too. Providing
62  * a custom configuration overrides the above selection.
63  *
64  * #define BSP_CONFIG_CUSTOM_CLOCK_MHZ 2.1
65  * #define BSP_CONFIG_MSP430_CUSTOM_BCSCTL1 9
66  * #define BSP_CONFIG_MSP430_CUSTOM_DCOCTL ( 1 << 5 )
67  */
68 
69 
70 /* ------------------------------------------------------------------------------------------------
71  * Exported Clock Configuration
72  * ------------------------------------------------------------------------------------------------
73  */
74 #if (defined BSP_CONFIG_CUSTOM_CLOCK_MHZ) || \
75  (defined BSP_CONFIG_MSP430_CUSTOM_BCSCTL1) || \
76  (defined BSP_CONFIG_MSP430_CUSTOM_DCOCTL)
77 #define BSP_CONFIG_CLOCK_MHZ BSP_CONFIG_CUSTOM_CLOCK_MHZ
78 #define BSP_CONFIG_MSP430_BCSCTL1 BSP_CONFIG_MSP430_CUSTOM_BCSCTL1
79 #define BSP_CONFIG_MSP430_DCOCTL BSP_CONFIG_MSP430_CUSTOM_DCOCTL
80 #else
81 
82 #if (BSP_CONFIG_CLOCK_MHZ_SELECT == 1)
83 #define BSP_CONFIG_CLOCK_MHZ 1.0
84 #define BSP_CONFIG_MSP430_BCSCTL1 CALBC1_1MHZ /* factory calibrated value from flash */
85 #define BSP_CONFIG_MSP430_DCOCTL CALDCO_1MHZ /* factory calibrated value from flash */
86 #elif (BSP_CONFIG_CLOCK_MHZ_SELECT == 2)
87 #define BSP_CONFIG_CLOCK_MHZ 2.1
88 #define BSP_CONFIG_MSP430_BCSCTL1 9
89 #define BSP_CONFIG_MSP430_DCOCTL ((/* DCO = */ 1 ) << 5 )
90 #elif (BSP_CONFIG_CLOCK_MHZ_SELECT == 4)
91 #define BSP_CONFIG_CLOCK_MHZ 3.9
92 #define BSP_CONFIG_MSP430_BCSCTL1 10
93 #define BSP_CONFIG_MSP430_DCOCTL ((/* DCO = */ 5 ) << 5 )
94 #elif (BSP_CONFIG_CLOCK_MHZ_SELECT == 6)
95 #define BSP_CONFIG_CLOCK_MHZ 6.1
96 #define BSP_CONFIG_MSP430_BCSCTL1 12
97 #define BSP_CONFIG_MSP430_DCOCTL ((/* DCO = */ 4 ) << 5 )
98 #elif (BSP_CONFIG_CLOCK_MHZ_SELECT == 8)
99 #define BSP_CONFIG_CLOCK_MHZ 8.0
100 #define BSP_CONFIG_MSP430_BCSCTL1 CALBC1_8MHZ /* factory calibrated value from flash */
101 #define BSP_CONFIG_MSP430_DCOCTL CALDCO_8MHZ /* factory calibrated value from flash */
102 #elif (BSP_CONFIG_CLOCK_MHZ_SELECT == 10)
103 #define BSP_CONFIG_CLOCK_MHZ 10.3
104 #define BSP_CONFIG_MSP430_BCSCTL1 14
105 #define BSP_CONFIG_MSP430_DCOCTL ((/* DCO = */ 2 ) << 5 )
106 #elif (BSP_CONFIG_CLOCK_MHZ_SELECT == 12)
107 #define BSP_CONFIG_CLOCK_MHZ 12.0
108 #define BSP_CONFIG_MSP430_BCSCTL1 CALBC1_12MHZ /* factory calibrated value from flash */
109 #define BSP_CONFIG_MSP430_DCOCTL CALDCO_12MHZ /* factory calibrated value from flash */
110 #elif (BSP_CONFIG_CLOCK_MHZ_SELECT == 16)
111 #define BSP_CONFIG_CLOCK_MHZ 16.0
112 #define BSP_CONFIG_MSP430_BCSCTL1 CALBC1_16MHZ /* factory calibrated value from flash */
113 #define BSP_CONFIG_MSP430_DCOCTL CALDCO_16MHZ /* factory calibrated value from flash */
114 #else
115 #error "ERROR: Unsupported clock speed. Custom clock speeds are possible. See comments in code."
116 #endif
117 
118 #endif
119

122  */
123 #endif
