114 // *************************************************************************************************
115 // @fn adc12_single_conversion
116 // @brief Init ADC12. Do single conversion. Turn off ADC12.
117 // @param none
118 // @return none
119 // *************************************************************************************************
120 u16 adc12_single_conversion(u16 ref, u16 sht, u16 channel)
121 {
122  // Initialize the shared reference module
123  REFCTL0 |= REFMSTR + ref + REFON; // Enable internal reference (1.5V or 2.5V)
124 
125  // Initialize ADC12_A
126  ADC12CTL0 = sht + ADC12ON; // Set sample time
127  ADC12CTL1 = ADC12SHP; // Enable sample timer
128  ADC12MCTL0 = ADC12SREF_1 + channel; // ADC input channel
129  ADC12IE = 0x001; // ADC_IFG upon conv result-ADCMEMO
130 
131  // Wait 2 ticks (66us) to allow internal reference to settle
132  Timer0_A4_Delay(2);
133 
134  // Start ADC12
135  ADC12CTL0 |= ADC12ENC;
136 
137  // Clear data ready flag
138  adc12_data_ready = 0;
139 
140  // Sampling and conversion start
141  ADC12CTL0 |= ADC12SC;
142 
143  // Wait until ADC12 has finished
144  Timer0_A4_Delay(5);
145  while (!adc12_data_ready);
146 
147  // Shut down ADC12
148  ADC12CTL0 &= ~(ADC12ENC | ADC12SC | sht);
149  ADC12CTL0 &= ~ADC12ON;
150 
151  // Shut down reference voltage
152  REFCTL0 &= ~(REFMSTR + ref + REFON);
153 
154  ADC12IE = 0;
155 
156  // Return ADC result
157  return (adc12_result);
158 }

163 // *************************************************************************************************
164 // @fn ADC12ISR
165 // @brief Store ADC12 conversion result. Set flag to indicate data ready.
166 // @param none
167 // @return none
168 // *************************************************************************************************
169 #pragma vector=ADC12_VECTOR
170 __interrupt void ADC12ISR (void)
171 {
172  switch(__even_in_range(ADC12IV,34))
173  {
174  case 0: break; // Vector 0: No interrupt
175  case 2: break; // Vector 2: ADC overflow
176  case 4: break; // Vector 4: ADC timing overflow
177  case 6: // Vector 6: ADC12IFG0
178  adc12_result = ADC12MEM0; // Move results, IFG is cleared
179  adc12_data_ready = 1;
180  _BIC_SR_IRQ(LPM3_bits); // Exit active CPU
181  break;
182  case 8: break; // Vector 8: ADC12IFG1
183  case 10: break; // Vector 10: ADC12IFG2
184  case 12: break; // Vector 12: ADC12IFG3
185  case 14: break; // Vector 14: ADC12IFG4
186  case 16: break; // Vector 16: ADC12IFG5
187  case 18: break; // Vector 18: ADC12IFG6
188  case 20: break; // Vector 20: ADC12IFG7
189  case 22: break; // Vector 22: ADC12IFG8
190  case 24: break; // Vector 24: ADC12IFG9
191  case 26: break; // Vector 26: ADC12IFG10
192  case 28: break; // Vector 28: ADC12IFG11
193  case 30: break; // Vector 30: ADC12IFG12
194  case 32: break; // Vector 32: ADC12IFG13
195  case 34: break; // Vector 34: ADC12IFG14
196  default: break;
197  }
198 }
