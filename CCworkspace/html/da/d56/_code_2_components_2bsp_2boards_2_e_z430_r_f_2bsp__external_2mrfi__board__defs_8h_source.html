58  */
59 #define __mrfi_GDO0_BIT__ 6
60 #define MRFI_CONFIG_GDO0_PIN_AS_INPUT() st( P2SEL &= ~BV(__mrfi_GDO0_BIT__); ) /* clear pin special function default */
61 #define MRFI_GDO0_PIN_IS_HIGH() (P2IN & BV(__mrfi_GDO0_BIT__))
62 
63 #define MRFI_GDO0_INT_VECTOR PORT2_VECTOR
64 #define MRFI_ENABLE_GDO0_INT() st( P2IE |= BV(__mrfi_GDO0_BIT__); ) /* atomic operation */
65 #define MRFI_DISABLE_GDO0_INT() st( P2IE &= ~BV(__mrfi_GDO0_BIT__); ) /* atomic operation */
66 #define MRFI_GDO0_INT_IS_ENABLED() ( P2IE & BV(__mrfi_GDO0_BIT__) )
67 #define MRFI_CLEAR_GDO0_INT_FLAG() st( P2IFG &= ~BV(__mrfi_GDO0_BIT__); ) /* atomic operation */
68 #define MRFI_GDO0_INT_FLAG_IS_SET() ( P2IFG & BV(__mrfi_GDO0_BIT__) )
69 #define MRFI_CONFIG_GDO0_RISING_EDGE_INT() st( P2IES &= ~BV(__mrfi_GDO0_BIT__); ) /* atomic operation */
70 #define MRFI_CONFIG_GDO0_FALLING_EDGE_INT() st( P2IES |= BV(__mrfi_GDO0_BIT__); ) /* atomic operation */
