40 /* Frame field defines and masks. Mask name must be field name with '_MSK' appended
41  * so the GET and PUT macros work correctly -- they use token pasting. Offset values
42  * are with respect to the MRFI payload and not the entire frame.
43  */
44 #define F_PORT_OS 0
45 #define F_PORT_OS_MSK (0x3F)
46 #define F_ENCRYPT_OS 0
47 #define F_ENCRYPT_OS_MSK (0x40)
48 #define F_FWD_FRAME 0
49 #define F_FWD_FRAME_MSK (0x80)
50 #define F_RX_TYPE 1
51 #define F_RX_TYPE_MSK (0x40)
52 #define F_ACK_REQ 1
53 #define F_ACK_REQ_MSK (0x80)
54 #define F_ACK_RPLY 1
55 #define F_ACK_RPLY_MSK (0x08)
56 #define F_TX_DEVICE 1
57 #define F_TX_DEVICE_MSK (0x30)
58 #define F_HOP_COUNT 1
59 #define F_HOP_COUNT_MSK (0x07)
60 #define F_TRACTID_OS 2
61 #define F_TRACTID_OS_MSK (0xFF)

97 /* macro to get a field from a frame buffer */
98 #define GET_FROM_FRAME(b,f) ((b)[f] & (f##_MSK))
99 
100 /* Macro to put a value 'v' into a frame buffer 'b'. 'v' value must already be shifted
101  * if necessary. 'b' is a byte array
102  */
103 #define PUT_INTO_FRAME(b,f,v) do {(b)[f] = ((b)[f] & ~(f##_MSK)) | (v); } while(0)
