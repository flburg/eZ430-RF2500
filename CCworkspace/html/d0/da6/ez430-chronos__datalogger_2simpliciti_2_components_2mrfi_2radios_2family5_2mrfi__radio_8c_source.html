379  *

381  **************************************************************************************************
382  */
383 void MRFI_Init(void)
384 {
385  /* ------------------------------------------------------------------
386  * Radio power-up reset
387  * ----------------------
388  */
389  memset(&mrfiIncomingPacket, 0x0, sizeof(mrfiIncomingPacket));
390 
391  /* Initialize the radio interface */
392  mrfiRadioInterfaceInit();
393 
394  /* Strobe Reset: Resets the radio and puts it in SLEEP state. */
395  MRFI_STROBE( SRES );
396 
397  /* verify the correct radio is installed */
398  MRFI_ASSERT( MRFI_RADIO_REG_READ( PARTNUM ) == MRFI_RADIO_PARTNUM ); /* incorrect radio specified */
399  MRFI_ASSERT( MRFI_RADIO_REG_READ( VERSION ) == MRFI_RADIO_VERSION ); /* incorrect radio specified */
400 
401  /* Put radio in Idle state */
402  MRFI_STROBE_IDLE_AND_WAIT();
403 
404 
405  /* ------------------------------------------------------------------
406  * Configure radio
407  * -----------------
408  */
409 
410  /* Configure Radio interrupts:
411  *
412  * RF1AIN_0 => Programmed to PA_PD signal.
413  * Configure it to interrupt on falling edge.
414  *
415  * RF1AIN_1 => Programmed to RSSI Valid signal.
416  * No need to configure for interrupt. This value will be read
417  * through polling.
418  *
419  * RF1AIN_9 => Rising edge indicates SYNC sent/received and
420  * Falling edge indicates end of packet.
421  * Configure it to interrupt on falling edge.
422  */
423 
424  /* Select Interrupt edge for PA_PD and SYNC signal:
425  * Interrupt Edge select register: 1 == Interrupt on High to Low transition.
426  */
427  RF1AIES = BV(0) | BV(9);
428 
429  /* Write the power output to the PA_TABLE and verify the write operation. */
430  {
431  uint8_t readbackPATableValue = 0;
432  bspIState_t s;
433 
434  BSP_ENTER_CRITICAL_SECTION(s);
435 
436  while( !(RF1AIFCTL1 & RFINSTRIFG));
437  RF1AINSTRW = 0x7E51; /* PA Table write (burst) */
438 
439  while( !(RF1AIFCTL1 & RFINSTRIFG));
440  RF1AINSTRB = RF_SNOP; /* reset pointer */
441 
442  while( !(RF1AIFCTL1 & RFINSTRIFG));
443  RF1AINSTRB = 0xFE; /* PA Table read (burst) */
444 
445  while( !(RF1AIFCTL1 & RFDINIFG));
446  RF1ADINB = 0x00; /* dummy write */
447 
448  while( !(RF1AIFCTL1 & RFDOUTIFG));
449  readbackPATableValue = RF1ADOUT0B;
450 
451  MRFI_ASSERT(readbackPATableValue == 0x51);
452 
453  while( !(RF1AIFCTL1 & RFINSTRIFG));
454  RF1AINSTRB = RF_SNOP;
455 
456  BSP_EXIT_CRITICAL_SECTION(s);
457  }
458

462

467  }

477  }
478  }

513  Mrfi_RxModeOff();

516  /* Chip bug: Radio does not come out of this SLEEP when put to sleep
517  * using the SPWD cmd. However, it does wakes up if SXOFF was used to
518  * put it to sleep.
519  */
520  MRFI_STROBE( SXOFF );
521 
522  /* Initial radio state is IDLE state */
523  mrfiRadioState = MRFI_RADIO_STATE_OFF;
524 
525  /*****************************************************************************************
526  * Compute reply delay scalar
527  *
528  * Formula from data sheet for all the narrow band radios is:
529  *
530  * (256 + DATAR_Mantissa) * 2^(DATAR_Exponent)
531  * DATA_RATE = ------------------------------------------ * f(xosc)
532  * 2^28
533  *
534  * To try and keep some accuracy we change the exponent of the denominator
535  * to (28 - (exponent from the configuration register)) so we do a division
536  * by a smaller number. We find the power of 2 by shifting.
537  *
538  * The maximum delay needed depends on the MAX_APP_PAYLOAD parameter. Figure
539  * out how many bits that will be when overhead is included. Bits/bits-per-second
540  * is seconds to transmit (or receive) the maximum frame. We multiply this number
541  * by 1000 to find the time in milliseconds. We then additionally multiply by
542  * 10 so we can add 5 and divide by 10 later, thus rounding up to the number of
543  * milliseconds. This last won't matter for slow transmissions but for faster ones
544  * we want to err on the side of being conservative and making sure the radio is on
545  * to receive the reply. The semaphore monitor will shut it down. The delay adds in
546  * a fudge factor that includes processing time on peer plus lags in Rx and processing
547  * time on receiver's side.
548  *
549  * Note that we assume a 26 MHz clock for the radio...
550  * ***************************************************************************************
551  */
552 #define MRFI_RADIO_OSC_FREQ 26000000
553 #define PHY_PREAMBLE_SYNC_BYTES 8
554 
555  {
556  uint32_t dataRate, bits;
557  uint16_t exponent, mantissa;
558 
559  /* mantissa is in MDMCFG3 */
560  mantissa = 256 + SMARTRF_SETTING_MDMCFG3;
561 
562  /* exponent is lower nibble of MDMCFG4. */
563  exponent = 28 - (SMARTRF_SETTING_MDMCFG4 & 0x0F);
564 
565  /* we can now get data rate */
566  dataRate = mantissa * (MRFI_RADIO_OSC_FREQ>>exponent);
567 
568  bits = ((uint32_t)((PHY_PREAMBLE_SYNC_BYTES + MRFI_MAX_FRAME_SIZE)*8))*10000;
569 
570  /* processing on the peer + the Tx/Rx time plus more */
571  sReplyDelayScalar = PLATFORM_FACTOR_CONSTANT + (((bits/dataRate)+5)/10);
572 
573  /* This helper value is used to scale the backoffs during CCA. At very
574  * low data rates we need to backoff longer to prevent continual sampling
575  * of valid frames which take longer to send at lower rates. Use the scalar
576  * we just calculated divided by 32. With the backoff algorithm backing
577  * off up to 16 periods this will result in waiting up to about 1/2 the total
578  * scalar value. For high data rates this does not contribute at all. Value
579  * is in microseconds.
580  */
581  sBackoffHelper = MRFI_BACKOFF_PERIOD_USECS + (sReplyDelayScalar>>5)*1000;
582  }
583

585  memset(mrfiIncomingPacket.frame, 0x00, sizeof(mrfiIncomingPacket.frame));
586  memset(mrfiIncomingPacket.rxMetrics, 0x00, sizeof(mrfiIncomingPacket.rxMetrics));
587

589  BSP_ENABLE_INTERRUPTS();

593 /**************************************************************************************************
594  * @fn MRFI_Transmit
595  *
596  * @brief Transmit a packet using CCA algorithm.
597  *
598  * @param pPacket - pointer to packet to transmit
599  *
600  * @return Return code indicates success or failure of transmit:
601  * MRFI_TX_RESULT_SUCCESS - transmit succeeded
602  * MRFI_TX_RESULT_FAILED - transmit failed because CCA failed
603  **************************************************************************************************
604  */
605 uint8_t MRFI_Transmit(mrfiPacket_t * pPacket, uint8_t txType)
606 {
607  uint8_t ccaRetries;
608  uint8_t txBufLen;
609  uint8_t returnValue = MRFI_TX_RESULT_SUCCESS;
610 
611  /* radio must be awake to transmit */
612  MRFI_ASSERT( mrfiRadioState != MRFI_RADIO_STATE_OFF );
613 
614  /* Turn off reciever. We can ignore/drop incoming packets during transmit. */
615  Mrfi_RxModeOff();
616 
617  /* compute number of bytes to write to transmit FIFO */
618  txBufLen = pPacket->frame[MRFI_LENGTH_FIELD_OFS] + MRFI_LENGTH_FIELD_SIZE;
619 
620  /* ------------------------------------------------------------------
621  * Write packet to transmit FIFO
622  * --------------------------------
623  */
624  MRFI_RADIO_WRITE_TX_FIFO(&(pPacket->frame[0]), txBufLen);
625 
626 
627  /* ------------------------------------------------------------------
628  * Immediate transmit
629  * ---------------------
630  */
631  if (txType == MRFI_TX_TYPE_FORCED)
632  {
633  /* Issue the TX strobe. */
634  MRFI_STROBE( STX );
635 
636  /* Wait for transmit to complete */
637  while(!MRFI_SYNC_PIN_INT_FLAG_IS_SET());
638 
639  /* Clear the interrupt flag */
640  MRFI_CLEAR_SYNC_PIN_INT_FLAG();
641  }
642  else
643  {
644  /* ------------------------------------------------------------------
645  * CCA transmit
646  * ---------------
647  */
648 
649  MRFI_ASSERT( txType == MRFI_TX_TYPE_CCA );
650 
651  /* set number of CCA retries */
652  ccaRetries = MRFI_CCA_RETRIES;
653 
654 
655  /* ===============================================================================
656  * Main Loop
657  * =============
658  */
659  for (;;)
660  {
661  /* Radio must be in RX mode for CCA to happen.
662  * Otherwise it will transmit without CCA happening.
663  */
664 
665  /* Can not use the Mrfi_RxModeOn() function here since it turns on the
666  * Rx interrupt, which we don't want in this case.
667  */
668  MRFI_STROBE( SRX );
669 
670  /* wait for the rssi to be valid. */
671  MRFI_RSSI_VALID_WAIT();
672 
673  /*
674  * Clear the PA_PD pin interrupt flag. This flag, not the interrupt itself,
675  * is used to capture the transition that indicates a transmit was started.
676  * The pin level cannot be used to indicate transmit success as timing may
677  * prevent the transition from being detected. The interrupt latch captures
678  * the event regardless of timing.
679  */
680  MRFI_CLEAR_PAPD_PIN_INT_FLAG();
681 
682  /* send strobe to initiate transmit */
683  MRFI_STROBE( STX );
684 
685  /* Delay long enough for the PA_PD signal to indicate a
686  * successful transmit. This is the 250 XOSC periods
687  * (9.6 us for a 26 MHz crystal).
688  * Found out that we need a delay of atleast 25 us on CC1100 to see
689  * the PA_PD signal change. Hence keeping the same for CC430
690  */
691  Mrfi_DelayUsec(25);
692 
693 
694  /* PA_PD signal goes from HIGH to LOW when going from RX to TX state.
695  * This transition is trapped as a falling edge interrupt flag
696  * to indicate that CCA passed and the transmit has started.
697  */
698  if (MRFI_PAPD_INT_FLAG_IS_SET())
699  {
700  /* ------------------------------------------------------------------
701  * Clear Channel Assessment passed.
702  * ----------------------------------
703  */
704 
705  /* Clear the PA_PD int flag */
706  MRFI_CLEAR_PAPD_PIN_INT_FLAG();
707 
708  /* PA_PD signal stays LOW while in TX state and goes back to HIGH when
709  * the radio transitions to RX state.
710  */
711  /* wait for transmit to complete */
712  while (!MRFI_PAPD_PIN_IS_HIGH());
713 
714  /* transmit done, break */
715  break;
716  }
717  else
718  {
719  /* ------------------------------------------------------------------
720  * Clear Channel Assessment failed.
721  * ----------------------------------
722  */
723 
724  /* Turn off radio and save some power during backoff */
725 
726  /* NOTE: Can't use Mrfi_RxModeOff() - since it tries to update the
727  * sync signal status which we are not using during the TX operation.
728  */
729  MRFI_STROBE_IDLE_AND_WAIT();
730 
731  /* flush the receive FIFO of any residual data */
732  MRFI_STROBE( SFRX );
733 
734  /* Retry ? */
735  if (ccaRetries != 0)
736  {
737  /* delay for a random number of backoffs */
738  Mrfi_RandomBackoffDelay();
739 
740  /* decrement CCA retries before loop continues */
741  ccaRetries--;
742  }
743  else /* No CCA retries are left, abort */
744  {
745  /* set return value for failed transmit and break */
746  returnValue = MRFI_TX_RESULT_FAILED;
747  break;
748  }
749  } /* CCA Failed */
750  } /* CCA loop */
751  }/* txType is CCA */
752 
753 
754  /* Done with TX. Clean up time... */
755 
756  /* Radio is already in IDLE state */
757 
758  /*
759  * Flush the transmit FIFO. It must be flushed so that
760  * the next transmit can start with a clean slate.
761  */
762  MRFI_STROBE( SFTX );
763 
764  /* If the radio was in RX state when transmit was attempted,
765  * put it back to Rx On state.
766  */
767  if(mrfiRadioState == MRFI_RADIO_STATE_RX)
768  {
769  Mrfi_RxModeOn();
770  }
771 
772  return( returnValue );
773 }

792 /**************************************************************************************************
793  * @fn Mrfi_SyncPinRxIsr
794  *
795  * @brief This interrupt is called when the SYNC signal transition from high to low.
796  * The sync signal is routed to the sync pin which is a GPIO pin. This high-to-low
797  * transition signifies a receive has completed. The SYNC signal also goes from
798  * high to low when a transmit completes. This is protected against within the
799  * transmit function by disabling sync pin interrupts until transmit completes.
800  *
801  * @param none
802  *
803  * @return none
804  **************************************************************************************************
805  */
806 static void Mrfi_SyncPinRxIsr(void)
807 {
808  uint8_t frameLen = 0x00;
809  uint8_t rxBytes;
810 
811  /* We should receive this interrupt only in RX state
812  * Should never receive it if RX was turned On only for
813  * some internal mrfi processing like - during CCA.
814  * Otherwise something is terribly wrong.
815  */
816  MRFI_ASSERT( mrfiRadioState == MRFI_RADIO_STATE_RX );
817 
818  /* ------------------------------------------------------------------
819  * Get RXBYTES
820  * -------------
821  */
822 
823  /*
824  * Read the RXBYTES register from the radio.
825  * Bit description of RXBYTES register:
826  * bit 7 - RXFIFO_OVERFLOW, set if receive overflow occurred
827  * bits 6:0 - NUM_BYTES, number of bytes in receive FIFO
828  *
829  * Due a chip bug, the RXBYTES register must read the same value twice
830  * in a row to guarantee an accurate value.
831  */
832  {
833  uint8_t rxBytesVerify;
834 
835  rxBytesVerify = MRFI_RADIO_REG_READ( RXBYTES );
836 
837  do
838  {
839  rxBytes = rxBytesVerify;
840  rxBytesVerify = MRFI_RADIO_REG_READ( RXBYTES );
841  }
842  while (rxBytes != rxBytesVerify);
843  }
844 
845 
846  /* ------------------------------------------------------------------
847  * FIFO empty?
848  * -------------
849  */
850 
851  /*
852  * See if the receive FIFIO is empty before attempting to read from it.
853  * It is possible nothing the FIFO is empty even though the interrupt fired.
854  * This can happen if address check is enabled and a non-matching packet is
855  * received. In that case, the radio automatically removes the packet from
856  * the FIFO.
857  */
858  if (rxBytes == 0)
859  {
860  /* receive FIFO is empty - do nothing, skip to end */
861  }
862  else
863  {
864  /* receive FIFO is not empty, continue processing */
865 
866  /* ------------------------------------------------------------------
867  * Process frame length
868  * ----------------------
869  */
870 
871  /* read the first byte from FIFO - the packet length */
872  MRFI_RADIO_READ_RX_FIFO(&frameLen, MRFI_LENGTH_FIELD_SIZE);
873 
874 
875  /*
876  * Make sure that the frame length just read corresponds to number of bytes in the buffer.
877  * If these do not match up something is wrong.
878  *
879  * This can happen for several reasons:
880  * 1) Incoming packet has an incorrect format or is corrupted.
881  * 2) The receive FIFO overflowed. Overflow is indicated by the high
882  * bit of rxBytes. This guarantees the value of rxBytes value will not
883  * match the number of bytes in the FIFO for overflow condition.
884  * 3) Interrupts were blocked for an abnormally long time which
885  * allowed a following packet to at least start filling the
886  * receive FIFO. In this case, all received and partially received
887  * packets will be lost - the packet in the FIFO and the packet coming in.
888  * This is the price the user pays if they implement a giant
889  * critical section.
890  * 4) A failed transmit forced radio to IDLE state to flush the transmit FIFO.
891  * This could cause an active receive to be cut short.
892  *
893  * Also check the sanity of the length to guard against rogue frames.
894  */
895  if ((rxBytes != (frameLen + MRFI_LENGTH_FIELD_SIZE + MRFI_RX_METRICS_SIZE)) ||
896  ((frameLen + MRFI_LENGTH_FIELD_SIZE) > MRFI_MAX_FRAME_SIZE) ||
897  (frameLen < MRFI_MIN_SMPL_FRAME_SIZE)
898  )
899  {
900  bspIState_t s;
901  noFrame++;
902 
903  /* mismatch between bytes-in-FIFO and frame length */
904 
905  /*
906  * Flush receive FIFO to reset receive. Must go to IDLE state to do this.
907  * The critical section guarantees a transmit does not occur while cleaning up.
908  */
909  BSP_ENTER_CRITICAL_SECTION(s);
910  MRFI_STROBE_IDLE_AND_WAIT();
911  MRFI_STROBE( SFRX );
912  MRFI_STROBE( SRX );
913  BSP_EXIT_CRITICAL_SECTION(s);
914 
915  /* flush complete, skip to end */
916  }
917  else
918  {
919  /* bytes-in-FIFO and frame length match up - continue processing */
920 
921  /* ------------------------------------------------------------------
922  * Get packet
923  * ------------
924  */
925 
926  /* clean out buffer to help protect against spurious frames */
927  memset(mrfiIncomingPacket.frame, 0x00, sizeof(mrfiIncomingPacket.frame));
928 
929  /* set length field */
930  mrfiIncomingPacket.frame[MRFI_LENGTH_FIELD_OFS] = frameLen;
931 
932  /* get packet from FIFO */
933  MRFI_RADIO_READ_RX_FIFO(&(mrfiIncomingPacket.frame[MRFI_FRAME_BODY_OFS]), frameLen);
934 
935  /* get receive metrics from FIFO */
936  MRFI_RADIO_READ_RX_FIFO(&(mrfiIncomingPacket.rxMetrics[0]), MRFI_RX_METRICS_SIZE);
937 
938 
939  /* ------------------------------------------------------------------
940  * CRC check
941  * ------------
942  */
943 
944  /*
945  * Note! Automatic CRC check is not, and must not, be enabled. This feature
946  * flushes the *entire* receive FIFO when CRC fails. If this feature is
947  * enabled it is possible to be reading from the FIFO and have a second
948  * receive occur that fails CRC and automatically flushes the receive FIFO.
949  * This could cause reads from an empty receive FIFO which puts the radio
950  * into an undefined state.
951  */
952 
953  /* determine if CRC failed */
954  if (!(mrfiIncomingPacket.rxMetrics[MRFI_RX_METRICS_CRC_LQI_OFS] & MRFI_RX_METRICS_CRC_OK_MASK))
955  {
956  /* CRC failed - do nothing, skip to end */
957  crcFail++;
958  }
959  else
960  {
961  /* CRC passed - continue processing */
962  crcPass++;
963 
964  /* ------------------------------------------------------------------
965  * Filtering
966  * -----------
967  */
968 
969  /* if address is not filtered, receive is successful */
970  if (!Mrfi_RxAddrIsFiltered(MRFI_P_DST_ADDR(&mrfiIncomingPacket)))
971  {
972  {
973  /* ------------------------------------------------------------------
974  * Receive successful
975  * --------------------
976  */
977 
978  /* Convert the raw RSSI value and do offset compensation for this radio */
979  mrfiIncomingPacket.rxMetrics[MRFI_RX_METRICS_RSSI_OFS] =
980  Mrfi_CalculateRssi(mrfiIncomingPacket.rxMetrics[MRFI_RX_METRICS_RSSI_OFS]);
981 
982  /* Remove the CRC valid bit from the LQI byte */
983  mrfiIncomingPacket.rxMetrics[MRFI_RX_METRICS_CRC_LQI_OFS] =
984  (mrfiIncomingPacket.rxMetrics[MRFI_RX_METRICS_CRC_LQI_OFS] & MRFI_RX_METRICS_LQI_MASK);
985 
986 
987  /* call external, higher level "receive complete" processing routine */
988  MRFI_RxCompleteISR();
989  }
990  }
991  }
992  }
993  }
994 
995  /* ------------------------------------------------------------------
996  * End of function
997  * -------------------
998  */
999 }

1202  /* Operand overwrite error */

1275  * Just having the Radio ON is not enough to read
1276  * the correct RSSI value. The Radio must in RX mode for

1290 /**************************************************************************************************
1291  * @fn Mrfi_CalculateRssi
1292  *
1293  * @brief Does binary to decimal conversion and offset compensation.
1294  *
1295  * @param none
1296  *
1297  * @return RSSI value in units of dBm.
1298  **************************************************************************************************
1299  */
1300 int8_t Mrfi_CalculateRssi(uint8_t rawValue)
1301 {
1302  int16_t rssi;
1303 
1304  /* The raw value is in 2's complement and in half db steps. Convert it to
1305  * decimal taking into account the offset value.
1306  */
1307  if(rawValue >= 128)
1308  {
1309  rssi = (int16_t)(rawValue - 256)/2 - MRFI_RSSI_OFFSET;
1310  }
1311  else
1312  {
1313  rssi = (rawValue/2) - MRFI_RSSI_OFFSET;
1314  }
1315 
1316  /* Restrict this value to least value can be held in an 8 bit signed int */
1317  if(rssi < -128)
1318  {
1319  rssi = -128;
