65 /**************************************************************************************************
66  * @fn SetVCore
67  *
68  * @brief SetVCore level is called from the user API. Change level by one step only.
69  *
70  * @param level - VcCore level
71  *
72  * @return none
73  **************************************************************************************************
74  */
75 static void SetVCore (uint8_t level)
76 {
77  uint8_t actLevel;
78  do {
79  actLevel = PMMCTL0_L & PMMCOREV_3;
80  if (actLevel < level)
81  SetVCoreUp(++actLevel); /* Set VCore (step by step) */
82  if (actLevel > level)
83  SetVCoreDown(--actLevel); /* Set VCore (step by step) */
84  }while (actLevel != level);
85 }

87 /**************************************************************************************************
88  * @fn SetVCoreUp
89  *
90  * @brief Set VCore up. Change level by one step only.
91  *
92  * @param level - VcCore level
93  *
94  * @return none
95  *************************************************************************************************
96  */
97 static void SetVCoreUp (uint8_t level)
98 {
99  PMMCTL0_H = 0xA5; /* Open PMM module registers for write access */
100 
101  SVSMHCTL = SVSHE + SVSHRVL0 * level + SVMHE + SVSMHRRL0 * level; /* Set SVS/M high side to new level */
102 
103  SVSMLCTL = SVSLE + SVMLE + SVSMLRRL0 * level; /* Set SVM new Level */
104  while ((PMMIFG & SVSMLDLYIFG) == 0); /* Wait till SVM is settled (Delay) */
105  PMMCTL0_L = PMMCOREV0 * level; /* Set VCore to x */
106  PMMIFG &= ~(SVMLVLRIFG + SVMLIFG); /* Clear already set flags */
107  if ((PMMIFG & SVMLIFG))
108  while ((PMMIFG & SVMLVLRIFG) == 0); /* Wait till level is reached */
109 
110  SVSMLCTL = SVSLE + SVSLRVL0 * level + SVMLE + SVSMLRRL0 * level; /* Set SVS/M Low side to new level */
111  PMMCTL0_H = 0x00; /* Lock PMM module registers for write access */
112 }
113 
114 /**************************************************************************************************
115  * @fn SetVCoreDown
116  *
117  * @brief Set VCore down
118  *
119  * @param level - VcCore level
120  *
121  * @return none
122  **************************************************************************************************
123  */
124 static void SetVCoreDown (uint8_t level)
125 {
126  PMMCTL0_H = 0xA5; /* Open PMM module registers for write access */
127  SVSMLCTL = SVSLE + SVSLRVL0 * level + SVMLE + SVSMLRRL0 * level; /* Set SVS/M Low side to new level */
128  while ((PMMIFG & SVSMLDLYIFG) == 0); /* Wait till SVM is settled (Delay) */
129  PMMCTL0_L = (level * PMMCOREV0); /* Set VCore to 1.85 V for Max Speed. */
130  PMMCTL0_H = 0x00; /* Lock PMM module registers for write access */
131 }

148 /**************************************************************************************************
149  * @fn Bsp_SetClocks
150  *
151  * @brief Set up system clocks.
152  *
153  * @param none
154  *
155  * @return none
156  **************************************************************************************************
157  */
158 static void Bsp_SetClocks(void)
159 {
160  /* Configure CPU clock for 12MHz */
161 
162  /* If clock settings are changed, remember to update BSP_TIMER_CLK_MHZ.
163  * Otherwise, all timer settings would be incorrect.
164  */
165  UCSCTL3 |= SELREF_2; /* Set DCO FLL reference = REFO */
166  UCSCTL4 |= SELA_2; /* Set ACLK = REFO */
167 
168  __bis_SR_register(SCG0); /* Disable the FLL control loop */
169  UCSCTL0 = 0x0000; /* Set lowest possible DCOx, MODx */
170  UCSCTL1 = DCORSEL_5; /* Select DCO range 24MHz operation */
171  UCSCTL2 = FLLD_1 + 374; /* Set DCO Multiplier for 12MHz */
172  /* (N + 1) * FLLRef = Fdco */
173  /* (374 + 1) * 32768 = 12MHz */
174  /* Set FLL Div = fDCOCLK/2 */
175  __bic_SR_register(SCG0); /* Enable the FLL control loop */
176 
177  /* Worst-case settling time for the DCO when the DCO range bits have been
178  * changed is n x 32 x 32 x f_MCLK / f_FLL_reference.
179  * 32 x 32 x 12 MHz / 32,768 Hz = 375000 = MCLK cycles for DCO to settle
180  */
181  __delay_cycles(375000);
182 
183  /* Loop until XT1,XT2 & DCO fault flag is cleared */
184  do
185  {
186  UCSCTL7 &= ~(XT2OFFG + XT1LFOFFG + DCOFFG);
187  /* Clear XT2,XT1,DCO fault flags */
188  SFRIFG1 &= ~OFIFG; /* Clear fault flags */
189  }while (SFRIFG1&OFIFG); /* Test oscillator fault flag */
190 
191  /* Select REFO as ACLK source and DCOCLK as MCLK and SMCLK source */
192  UCSCTL4 = SELA__REFOCLK | SELS__DCOCLKDIV | SELM__DCOCLKDIV;
193 }

195 /**************************************************************************************************
196  * @fn BSP_EARLY_INIT
197  *
198  * @brief This function is called by start-up code before doing the normal initialization
199  * of data segments. If the return value is zero, initialization is not performed.
200  * The global macro label "BSP_EARLY_INIT" gets #defined in the bsp_msp430_defs.h
201  * file, according to the specific compiler environment (CCE or IAR). In the CCE
202  * environment this macro invokes "_system_pre_init()" and in the IAR environment
203  * this macro invokes "__low_level_init()".
204  *
205  * @param None
206  *
207  * @return 0 - don't intialize data segments / 1 - do initialization
208  *************************************************************************************************
209  */
210 BSP_EARLY_INIT(void)
211 {
212  /* Disable watchdog timer */
213  WDTCTL = WDTPW | WDTHOLD;
214 
215  /* Setup Vcore */
216  Bsp_SetVCore();
217 
218  /* Configure System clocks */
219  Bsp_SetClocks();
220 
221  /* Return 1 - run seg_init */
222  return (1);
223 }
