66 /* Network header size definition */
67 /* ********************************* NOTE ****************************************
68  * There is a dependency here that really shouldn't exist. A reimplementation
69  * is necessary to avoid it.
70  *
71  * MRFI allocates the frame buffer which means it needs to know at compile time
72  * how big the buffer is. This means in must know the NWK header size, the
73  * maximum NWK and User application payload sizes and whether Security is enabled.
74  * ********************************************************************************
75  */
76 #ifndef SMPL_SECURE
77 #define NWK_HDR_SIZE 3
78 #define NWK_PAYLOAD MAX_NWK_PAYLOAD
79 #else
80 #define NWK_HDR_SIZE 6
81 #define NWK_PAYLOAD (MAX_NWK_PAYLOAD+4)
82 #endif
83 
84 /* if external code has defined a maximum payload, use that instead of default */
85 #ifdef MAX_APP_PAYLOAD
86 #ifndef MAX_NWK_PAYLOAD
87 #error ERROR: MAX_NWK_PAYLOAD not defined
88 #endif
89 #if MAX_APP_PAYLOAD < NWK_PAYLOAD
90 #define MAX_PAYLOAD NWK_PAYLOAD
91 #else
92 #define MAX_PAYLOAD MAX_APP_PAYLOAD
93 #endif
94 #define MRFI_MAX_PAYLOAD_SIZE (MAX_PAYLOAD+NWK_HDR_SIZE) /* SimpliciTI payload size plus six byte overhead */
95 #endif
