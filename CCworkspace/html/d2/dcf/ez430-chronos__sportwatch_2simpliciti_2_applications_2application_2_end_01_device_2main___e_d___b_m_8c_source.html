69 // *************************************************************************************************
70 // @fn simpliciti_link
71 // @brief Init hardware and try to link to access point.
72 // @param none
73 // @return unsigned char 0 = Could not link, timeout or external cancel.
74 // 1 = Linked successful.
75 // *************************************************************************************************
76 unsigned char simpliciti_link(void)
77 {
78  uint8_t timeout;
79  addr_t lAddr;
80  uint8_t i;
81  uint8_t pwr;
82 
83  // Configure timer
84  BSP_InitBoard();
85 
86  // Change network address to value set in calling function
87  for (i=0; i<NET_ADDR_SIZE; i++)
88  {
89  lAddr.addr[i] = simpliciti_ed_address[i];
90  }
91  SMPL_Ioctl(IOCTL_OBJ_ADDR, IOCTL_ACT_SET, &lAddr);
92 
93  // Set flag
94  simpliciti_flag = SIMPLICITI_STATUS_LINKING;
95 
96  /* Keep trying to join (a side effect of successful initialization) until
97  * successful. Toggle LEDS to indicate that joining has not occurred.
98  */
99  timeout = 0;
100  while (SMPL_SUCCESS != SMPL_Init(0))
101  {
102  NWK_DELAY(1000);
103 
104  // Service watchdog
105  WDTCTL = WDTPW + WDTIS__512K + WDTSSEL__ACLK + WDTCNTCL;
106 
107  // Stop connecting after defined numbers of seconds (15)
108  if (timeout++ > TIMEOUT)
109  {
110  // Clean up SimpliciTI stack to enable restarting
111  sInit_done = 0;
112  simpliciti_flag = SIMPLICITI_STATUS_ERROR;
113  return (0);
114  }
115 
116  // Break when flag bit SIMPLICITI_TRIGGER_STOP is set
117  if (getFlag(simpliciti_flag, SIMPLICITI_TRIGGER_STOP))
118  {
119  // Clean up SimpliciTI stack to enable restarting
120  sInit_done = 0;
121  return (0);
122  }
123  }
124 
125  // Set output power to +3.3dmB
126  pwr = IOCTL_LEVEL_2;
127  SMPL_Ioctl(IOCTL_OBJ_RADIO, IOCTL_ACT_RADIO_SETPWR, &pwr);
128 
129  /* Unconditional link to AP which is listening due to successful join. */
130  timeout = 0;
131  while (SMPL_SUCCESS != SMPL_Link(&sLinkID1))
132  {
133  NWK_DELAY(1000);
134 
135  // Service watchdog
136  WDTCTL = WDTPW + WDTIS__512K + WDTSSEL__ACLK + WDTCNTCL;
137 
138  // Stop linking after timeout
139  if (timeout++ > TIMEOUT)
140  {
141  // Clean up SimpliciTI stack to enable restarting
142  sInit_done = 0;
143  simpliciti_flag = SIMPLICITI_STATUS_ERROR;
144  return (0);
145  }
146 
147  // Exit when flag bit SIMPLICITI_TRIGGER_STOP is set
148  if (getFlag(simpliciti_flag, SIMPLICITI_TRIGGER_STOP))
149  {
150  // Clean up SimpliciTI stack to enable restarting
151  sInit_done = 0;
152  return (0);
153  }
154  }
155  simpliciti_flag = SIMPLICITI_STATUS_LINKED;
156 
157  return (1);
158 }

162 // *************************************************************************************************
163 // @fn simpliciti_main_tx_only
164 // @brief Get data through callback. Transfer data when external trigger is set.
165 // @param none
166 // @return none
167 // *************************************************************************************************
168 void simpliciti_main_tx_only(void)
169 {
170  while(1)
171  {
172  // Get end device data from callback function
173  simpliciti_get_ed_data_callback();
174 
175  // Send data when flag bit SIMPLICITI_TRIGGER_SEND_DATA is set
176  if (getFlag(simpliciti_flag, SIMPLICITI_TRIGGER_SEND_DATA))
177  {
178  // Get radio ready. Wakes up in IDLE state.
179  SMPL_Ioctl( IOCTL_OBJ_RADIO, IOCTL_ACT_RADIO_AWAKE, 0);
180 
181  // Acceleration / button events packets are 4 bytes long
182  SMPL_SendOpt(sLinkID1, simpliciti_data, 4, SMPL_TXOPTION_NONE);
183 
184  // Put radio back to SLEEP state
185  SMPL_Ioctl( IOCTL_OBJ_RADIO, IOCTL_ACT_RADIO_SLEEP, 0);
186 
187  clearFlag(simpliciti_flag, SIMPLICITI_TRIGGER_SEND_DATA);
188  }
189 
190  // Exit when flag bit SIMPLICITI_TRIGGER_STOP is set
191  if (getFlag(simpliciti_flag, SIMPLICITI_TRIGGER_STOP))
192  {
193  // Clean up SimpliciTI stack to enable restarting
194  sInit_done = 0;
195  break;
196  }
197  }
198 }

202 // *************************************************************************************************
203 // @fn simpliciti_main_sync
204 // @brief Send ready-to-receive packets in regular intervals. Listen shortly for host reply.
205 // Decode received host command and trigger action.
206 // @param none
207 // @return none
208 // *************************************************************************************************
209 void simpliciti_main_sync(void)
210 {
211  uint8_t len, i;
212  uint8_t ed_data[2];
213 
214  while(1)
215  {
216  // Sleep 0.5sec between ready-to-receive packets
217  // SimpliciTI has no low power delay function, so we have to use ours
218  Timer0_A4_Delay(CONV_MS_TO_TICKS(500));
219 
220  // Get radio ready. Radio wakes up in IDLE state.
221  SMPL_Ioctl( IOCTL_OBJ_RADIO, IOCTL_ACT_RADIO_AWAKE, 0);
222 
223  // Send 2 byte long ready-to-receive packet to stimulate host reply
224  ed_data[0] = SYNC_ED_TYPE_R2R;
225  ed_data[1] = 0xCB;
226  SMPL_SendOpt(sLinkID1, ed_data, 2, SMPL_TXOPTION_NONE);
227 
228  // Wait shortly for host reply
229  SMPL_Ioctl( IOCTL_OBJ_RADIO, IOCTL_ACT_RADIO_RXON, 0);
230  NWK_DELAY(10);
231 
232  // Check if a command packet was received
233  while (SMPL_Receive(sLinkID1, simpliciti_data, &len) == SMPL_SUCCESS)
234  {
235  // Decode received data
236  if (len > 0)
237  {
238  // Use callback function in application to decode data and react
239  simpliciti_sync_decode_ap_cmd_callback();
240 
241  // Get reply data and send out reply packet burst (19 bytes each)
242  for (i=0; i<simpliciti_reply_count; i++)
243  {
244  NWK_DELAY(10);
245  simpliciti_sync_get_data_callback(i);
246  SMPL_SendOpt(sLinkID1, simpliciti_data, BM_SYNC_DATA_LENGTH, SMPL_TXOPTION_NONE);
247  }
248  }
249  }
250 
251  // Put radio back to sleep
252  SMPL_Ioctl( IOCTL_OBJ_RADIO, IOCTL_ACT_RADIO_SLEEP, 0);
253 
254  // Service watchdog
255  WDTCTL = WDTPW + WDTIS__512K + WDTSSEL__ACLK + WDTCNTCL;
256 
257  // Exit when flag bit SIMPLICITI_TRIGGER_STOP is set
258  if (getFlag(simpliciti_flag, SIMPLICITI_TRIGGER_STOP))
259  {
260  // Clean up SimpliciTI stack to enable restarting
261  sInit_done = 0;
262  break;
263  }
264  }
