67 /**************************************************************************************************
68  * @fn BSP_EARLY_INIT
69  *
70  * @brief This function is called by start-up code before doing the normal initialization
71  * of data segments. If the return value is zero, initialization is not performed.
72  * The global macro label "BSP_EARLY_INIT" gets #defined in the bsp_msp430_defs.h
73  * file, according to the specific compiler environment (CCE or IAR). In the CCE
74  * environment this macro invokes "_system_pre_init()" and in the IAR environment
75  * this macro invokes "__low_level_init()".
76  *
77  * @param None
78  *
79  * @return 0 - don't intialize data segments / 1 - do initialization
80  **************************************************************************************************
81 */
82 BSP_EARLY_INIT(void)
83 {
84  /* Disable watchdog timer */
85  WDTCTL = WDTPW | WDTHOLD;
86 
87  /* Return 1 - run seg_init */
88  return (1);
89 }

91 /**************************************************************************************************
92  * @fn BSP_InitBoard
93  *
94  * @brief Initialize the board.
95  *
96  * @param none
97  *
98  * @return none
99  **************************************************************************************************
100  */
101 void BSP_InitBoard(void)
102 {
103  /* configure internal digitally controlled oscillator */
104  DCOCTL = BSP_CONFIG_MSP430_DCOCTL;
105  BCSCTL1 = BSP_CONFIG_MSP430_BCSCTL1;
106 
107  /* Configure TimerA for use by the delay function */
108 
109  /* Reset the timer */
110  TACTL |= TACLR; /* Set the TACLR */
111 
112  /* Clear all settings */
113  TACTL = 0x0;
114 
115  /* Select the clk source to be - SMCLK (Sub-Main CLK)*/
116  TACTL |= TASSEL_2;
117 
118 #if defined(SW_TIMER)
119 #define MHZ_CLOCKS_PER_USEC BSP_CLOCK_MHZ
120 #define MHZ_CLOCKS_PER_ITERATION 10
121 
122  sIterationsPerUsec = (uint8_t)(((MHZ_CLOCKS_PER_USEC)/(MHZ_CLOCKS_PER_ITERATION))+.5);
123 
124  if (!sIterationsPerUsec)
125  {
126  sIterationsPerUsec = 1;
127  }
128 #endif /* SW_TIMER */
129 }
130 
131 /**************************************************************************************************
132  * @fn BSP_Delay
133  *
134  * @brief Sleep for the requested amount of time.
135  *
136  * @param # of microseconds to sleep.
137  *
138  * @return none
139  **************************************************************************************************
140  */
141 void BSP_Delay(uint16_t usec)
142 #if !defined(SW_TIMER)
143 {
144 
145  TAR = 0; /* initial count */
146  TACCR0 = BSP_TIMER_CLK_MHZ*usec; /* compare count. (delay in ticks) */
147 
148  /* Start the timer in UP mode */
149  TACTL |= MC_1;
150 
151  /* Loop till compare interrupt flag is set */
152  while(!(TACCTL0 & CCIFG));
153 
154  /* Stop the timer */
155  TACTL &= ~(MC_1);
156 
157  /* Clear the interrupt flag */
158  TACCTL0 &= ~CCIFG;
159 }
160 
161 
162 #else /* !SW_TIMER */
163 
164 
165 {
166  /* Declared 'volatile' in case User optimizes for speed. This will
167  * prevent the optimizer from eliminating the loop completely. But
168  * it also generates more code...
169  */
170  volatile uint16_t repeatCount = sIterationsPerUsec*usec;
171 
172  while (repeatCount--) ;
173 
174  return;
175 }
176 
177 #endif /* !SW_TIMER */
