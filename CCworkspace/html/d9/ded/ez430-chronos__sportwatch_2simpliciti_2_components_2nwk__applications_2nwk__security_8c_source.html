50 /* *** GENERAL SECURITY OUTLINE ***
51  *
52  * We are using XTEA (eXtended Tiny Encryption Algorithm) with a fixed
53  * number of rounds (32). We have removed the parameters from the API
54  * we harvested from the public domain.
55  *
56  * We are using a CTR-like mode. We use the 64-bit block cipher function of the
57  * XTEA code to encipher a concatenation of the 32-bit initialization vector and
58  * a 32-bit counter that increments each block. We encrypt using a fixed 128-bit
59  * key. The resulting 64-bit output is XOR'ed with the message. If the message is
60  * longer than 64 bits we encipher the next block (incrementing the counter) and
61  * continue until the message is exhausted. If the last cipher block is longer
62  * than the message we simply discard the remaining cipher block.
63  */
64

107 /* 32-bit Initialization vector */
108 static uint32_t const sIV = 0x87654321;
109 
110 /* 128-bit (16 byte) key. Initialized as string but fetched and used in XTEA
111  * encryption as 4 unsigned longs. Endianess could count if the peers are on
112  * two different MCUs. Endianess is rectified in initialization code.
113  *
114  * Initialization _MUST_ be done as a string (or character array). Though it
115  * won't matter how the initialization is done if both peers are the same
116  * endianness, good prectice will initialize these as a string (or character
117  * array) so that the endianess reconciliation works properly for all cases.
118  */
119 static key_t sKey = {"SimpliciTI's Key"};
120 
121 /* Constant set as an authentication code. Note that since it is a
122  * fixed value as opposed to a hash of the message it does not provide
123  * an integrity check. It will only differentiate two message encryptions
124  * with the same LSB but different MSB components. Thus it helps guard
125  * against replays.
126  */
127 static secMAC_t const sMAC = 0xA5;

131  */

191 /******************************************************************************
192  * @fn msg_encipher
193  *
194  * @brief Encipher a message using the XTEA algorithm and the modified
195  * CTR mode method.
196  *
197  * input parameters
198  * @param msg - pointer to message to encipher
199  * @param len - length of message
200  * @param cntStart - pointer to the counter used in the cipher block.
201  *
202  * output parameters
203  * @param cntStart - counter is updated during encryption.
204  *
205  * @return void
206  */
207 #ifdef SMPL_SECURE
208 static void msg_encipher(uint8_t *msg, uint8_t len, uint32_t *cntStart)
209 {
210  uint8_t i, idx, done;
211  uint8_t *mptr = (uint8_t *)&sMsg[0];
212  uint32_t ctr;
213 
214  if ((NULL == msg) || !len)
215  {
216  return;
217  }
218 
219  /* set local counter from input */
220  ctr = *cntStart;
221 
222  idx = 0;
223  done = 0;
224  do
225  {
226  /* Set block to be enciphered. 1st 32 bits are the IV. The second
227  * 32 bits are the current CTR value.
228  */
229  sMsg[0] = sIV;
230  sMsg[1] = ctr;
231  /* encrypt */
232  xtea_encipher();
233  /* increment counter for next time. */
234  ctr++;
235  /* XOR ciphered block with message to be sent. Only operate
236  * up to and including the last message byte which may not
237  * be on a cipher block boundary (64 bits == 8 bytes).
238  */
239  for (i=0; i<sizeof(sMsg) && idx<len; ++i, ++idx)
240  {
241  msg[idx] ^= mptr[i];
242  }
243 
244  if (idx >= len)
245  {
246  /* we're done */
247  done = 1;
248  }
249  } while (!done);
250 
251  /* return counter value start for next time */
252  *cntStart = ctr;
253 
254  return;
255 }

280 /******************************************************************************
281  * @fn xtea_encipher
282  *
283  * @brief XTEA encipher algorithm. Calling arguments removed from public
284  * domain code and static-scope values used instead.
285  *
286  * input parameters
287  *
288  * output parameters
289  *
290  * @return void
291  */
292 void xtea_encipher(void)
293 {
294  uint32_t v0=sMsg[0], v1=sMsg[1];
295  uint16_t i;
296  uint32_t sum=0, delta=0x9E3779B9;
297 
298  for(i=0; i<NUM_ROUNDS; i++)
299  {
300  v0 += (((v1 << 4) ^ (v1 >> 5)) + v1) ^ (sum + sKey.keyL[sum & 3]);
301  sum += delta;
302  v1 += (((v0 << 4) ^ (v0 >> 5)) + v0) ^ (sum + sKey.keyL[(sum>>11) & 3]);
303  }
304 
305  sMsg[0]=v0;
306  sMsg[1]=v1;
307 }

309 /******************************************************************************
310  * @fn nwk_setSecureFrame
311  *
312  * @brief Called from NWK to secure a frame.
313  *
314  * input parameters
315  * @param frame - pointer to frame to secure
316  * @param msglen - length of message
317  * @param ctr - pointer to the counter used in the cipher block. This will
318  * be NULL if a network application is sending a frame. Since
319  * these are not connection-based there is no counter sync
320  * issue but we still need a counter value. A random value
321  * is used.
322  *
323  * output parameters
324  * @param cntStart - counter is updated during encryption.
325  *
326  * @return void
327  */
328 void nwk_setSecureFrame(mrfiPacket_t *frame, uint8_t msglen, uint32_t *ctr)
329 {
330  uint32_t locCnt;
331 
332  /* If an encrypted frame is to be sent to a non-connection based port use a
333  * random number as the lsb counter value. In this case only the lsb is used
334  * for a counter value during decryption. Not as secure but there are still
335  * the 32 bits in the IV.
336  */
337  locCnt = ctr ? *ctr : MRFI_RandomByte();
338 
339  /* place counter value into frame */
340  PUT_INTO_FRAME(MRFI_P_PAYLOAD(frame), F_SEC_CTR_OS, (uint8_t)(locCnt & 0xFF));
341 
342  /* Put MAC value in */
343  nwk_putNumObjectIntoMsg((void *)&sMAC, (void *)(MRFI_P_PAYLOAD(frame)+F_SEC_MAC_OS), sizeof(secMAC_t));
344 
345  /* Put FCS value in */
346  {
347  secFCS_t fcs = calcFCS(MRFI_P_PAYLOAD(frame)+F_SEC_MAC_OS, msglen+sizeof(secMAC_t));
348 
349  nwk_putNumObjectIntoMsg((void *)&fcs, (void *)(MRFI_P_PAYLOAD(frame)+F_SEC_ICHK_OS), sizeof(secFCS_t));
350  }
351 
352  /* Encrypt frame */
353  msg_encipher(MRFI_P_PAYLOAD(frame)+F_SEC_ICHK_OS, msglen+sizeof(secMAC_t)+sizeof(secFCS_t), &locCnt);
354 
355  /* Set the Encryption bit */
356  PUT_INTO_FRAME(MRFI_P_PAYLOAD(frame), F_ENCRYPT_OS, F_ENCRYPT_OS_MSK);
357 
358  /* Update the counter if it was a "real" counter. */
359  if (ctr)
360  {
361  *ctr = locCnt;
362  }
363 
364  return;
365 }

367 /******************************************************************************
368  * @fn calcFCS
369  *
370  * @brief Calculate the frame check sequence. Currently it's just a
371  * cumulative XOR of each byte starting with the MAC byte. The
372  * FCS is placed in front of the MAC after the counter hint and is
373  * included in the encryption.
374  *
375  * input parameters
376  * @param msg - pointer to message
377  * @param len - length of message
378  *
379  * output parameters
380  *
381  * @return Returns the FCS using the typedef.
382  */
383 static secFCS_t calcFCS(uint8_t *msg, uint8_t len)
384 {
385  uint8_t i;
386  secFCS_t result = 0;
387 
388  for (i=0; i<len; ++i)
389  {
390  result ^= *(msg+i);
391  }
392 
393  return result;
394 }

396 /******************************************************************************
397  * @fn nwk_getSecureFrame
398  *
399  * @brief Called from NWK to get a secure a frame and decrypt.
400  *
401  * input parameters
402  * @param frame - pointer to frame containing encrypted message
403  * @param msglen - length of message
404  * @param ctr - pointer to the counter used in the cipher block. This will
405  * be NULL if a network applicaiton is getting a frame. Since
406  * these are not connection-nbased there is no counter sync
407  * issue but we still need a counter value.
408  *
409  * output parameters
410  * @param cntStart - counter is updated during decryption. If decryption fails
411  * this value is not changed.
412  *
413  * @return Returns non-zero if frame decryption is valid, otherwise returns 0.
414  */
415 uint8_t nwk_getSecureFrame(mrfiPacket_t *frame, uint8_t msglen, uint32_t *ctr)
416 {
417  uint8_t rc = 1;
418  uint8_t done = 0;
419  uint8_t cntHint = GET_FROM_FRAME(MRFI_P_PAYLOAD(frame), F_SEC_CTR_OS);
420  uint32_t locCnt, frameCnt;
421 
422  /* Construct proposed CTR values */
423 
424  /* Just like encryption, we may be talking to a non-connection based
425  * peer in which case the counter value is represented by the lsb byte
426  * conveyed in the frame.
427  */
428  locCnt = ctr ? *ctr : cntHint;
429 
430  frameCnt = (locCnt & 0xFFFFFF00) + cntHint;
431 
432  do
433  {
434  /* See if counters match */
435  if (locCnt == frameCnt)
436  {
437  /* When the counters appear to match is the only time we actually decipher
438  * the message. It is the only time we can do so since out-of-sync lsb counter
439  * values guarantees that something is wrong somewhere. Decryption is successful
440  * only if the MAC and FCS values match. The message is left as-is after the
441  * decipher attempt. Either it appears valid or is doesn't and is discarded.
442  * There is no recovery attempt if the counters match but the MAC or FCS do
443  * not. It is considered a rogue message.
444  */
445  msg_decipher(MRFI_P_PAYLOAD(frame)+F_SEC_ICHK_OS, msglen-1, &locCnt);
446 
447  /* Get MAC and make sure it matches. A failure can occur if a replayed frame happens
448  * to have the correct counter sync value but was encoded with the wrong complete
449  * counter value. Otherwise the MAC values must match when the counter values are equal.
450  */
451  {
452  secMAC_t mac;
453 
454  nwk_getNumObjectFromMsg((void *)(MRFI_P_PAYLOAD(frame)+F_SEC_MAC_OS), (void *)&mac, sizeof(secMAC_t));
455  if (mac != sMAC)
456  {
457  rc = 0;
458  }
459  }
460 
461  /* FCS check... */
462  {
463  secFCS_t fcs;
464 
465  nwk_getNumObjectFromMsg((void *)(MRFI_P_PAYLOAD(frame)+F_SEC_ICHK_OS), (void *)&fcs, sizeof(secFCS_t));
466  if (fcs != calcFCS(MRFI_P_PAYLOAD(frame)+F_SEC_MAC_OS, msglen-1-sizeof(secMAC_t)))
467  {
468  rc = 0;
469  }
470  }
471 
472  /* we're done. */
473  done = 1;
474  }
475  else
476  {
477  /* Uh oh. Counters don't match. Try and resync. We need to distinguish among
478  * missed frames, duplicates and rogues plus account for counter wrap.
479  */
480  if (frameCnt > locCnt)
481  {
482  /* frameCnt is bigger. Second part of test below takes care of
483  * the unlikely case of a complete counter wrap (msb's all 0) in
484  * which case the test will incorrectly fail when the count is
485  * actually within the (wrapped) window. #ifdef'ed to avoid compiler
486  * warning in case user sets CNT_WINDOW to 0 (pointless comparison of
487  * unsigned value).
488  */
489  if (((frameCnt-CTR_WINDOW) <= locCnt)
490 #if CTR_WINDOW > 0
491  || (frameCnt < CTR_WINDOW)
492 #endif
493  )
494  {
495  /* Value within window. We probably missed something. Adjust and decipher.
496  * If locCnt is less because it wrapped and frameCnt didn't it means that
497  * it's a duplicate or late frame. In that case the following will lead to
498  * a decryption that fails sanity checks which is OK because the frame will
499  * be correctly rejected.
500  */
501  locCnt = frameCnt;
502  }
503  else
504  {
505  /* It's either a rogue or a really old duplicate packet. In either case
506  * we dismiss the frame.
507  */
508  rc = 0;
509  done = 1;
510  }
511  }
512  else
513  {
514  /* locCnt is bigger. The only way the frame can be valid is if the
515  * counter wrapped causing frameCnt to appear to be smaller. Wrap the
516  * counter and decrypt. If the frame isn't valid, i.e., it's late,
517  * a duplicate, or a rogue, the decryption will fail sanity checks and
518  * the frame will be correctly rejected. The following arithmetic works
519  * correctly without a special test for the complete counter wrap case.
520  */
521  frameCnt += 0x100; /* wrap the hint-based counter */
522  if (((frameCnt-CTR_WINDOW) <= locCnt))
523  {
524  /* An lsb wrap but still within window. We probably missed something.
525  * Adjust (with wrap) and decrypt.
526  */
527  locCnt = frameCnt;
528  }
529  else
530  {
531  /* rogue frame */
532  rc = 0;
533  done = 1;
534  }
535  }
536  }
537  } while (!done);
538 
539  if (ctr && rc)
540  {
541  /* Only update the counter if the count was a "real" one and the
542  * decryption succeeded.
543  */
544  *ctr = locCnt;
545  }
546 
547  return rc;
548 }
