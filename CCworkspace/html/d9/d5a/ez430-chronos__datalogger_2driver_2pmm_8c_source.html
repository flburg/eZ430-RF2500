17 // Set VCore level
18 // SetVCore level is called from the user API
19 //****************************************************************************//
20 void SetVCore (unsigned char level) // Note: change level by one step only
21 {
22  unsigned char actLevel;
23  do {
24  actLevel = PMMCTL0_L & PMMCOREV_3;
25  if (actLevel < level)
26  SetVCoreUp(++actLevel); // Set VCore (step by step)
27  if (actLevel > level)
28  SetVCoreDown(--actLevel); // Set VCore (step by step)
29  }while (actLevel != level);
30 }

34 // Set VCore up
35 //****************************************************************************//
36 void SetVCoreUp (unsigned char level) // Note: change level by one step only
37 {
38  PMMCTL0_H = 0xA5; // Open PMM module registers for write access
39 
40  SVSMHCTL = SVSHE + SVSHRVL0 * level + SVMHE + SVSMHRRL0 * level; // Set SVS/M high side to new level
41 
42  SVSMLCTL = SVSLE + SVMLE + SVSMLRRL0 * level; // Set SVM new Level
43  while ((PMMIFG & SVSMLDLYIFG) == 0); // Wait till SVM is settled (Delay)
44  PMMCTL0_L = PMMCOREV0 * level; // Set VCore to x
45  PMMIFG &= ~(SVMLVLRIFG + SVMLIFG); // Clear already set flags
46  if ((PMMIFG & SVMLIFG))
47  while ((PMMIFG & SVMLVLRIFG) == 0); // Wait till level is reached
48 
49  SVSMLCTL = SVSLE + SVSLRVL0 * level + SVMLE + SVSMLRRL0 * level; // Set SVS/M Low side to new level
50  PMMCTL0_H = 0x00; // Lock PMM module registers for write access
51 }

56 // Set VCore down
57 //****************************************************************************//
58 void SetVCoreDown (unsigned char level)
59 {
60  PMMCTL0_H = 0xA5; // Open PMM module registers for write access
61  SVSMLCTL = SVSLE + SVSLRVL0 * level + SVMLE + SVSMLRRL0 * level; // Set SVS/M Low side to new level
62  while ((PMMIFG & SVSMLDLYIFG) == 0); // Wait till SVM is settled (Delay)
63  PMMCTL0_L = (level * PMMCOREV0); // Set VCore to 1.85 V for Max Speed.
64  PMMCTL0_H = 0x00; // Lock PMM module registers for write access
65 }
