186 /***************************************************************************************
187  * ** NV Object support **
188  *
189  * The following object supports saving and restoring the information
190  * necessary to save and restore a device connection context.
191  *
192  * On a GET interface populates the IOCTL object with the version and length (in bytes)
193  * of the current static connection iformation area. In addition it populates the address
194  * pointed to by 'objPtr' with the base address of the connection context. At this point
195  * the caller can either copy to or from the address. Note that this is a dangerous
196  * interface, as the caller is provided with direct access to the connection context.
197  *
198  * When restoring the connection context some sanity checks are possible. If the
199  * version or length elements of the saved context do not match those of the current
200  * static object the static object should not be populated. If this sanity fails the
201  * caller is not provided with the pointer to the conneciton ocntext.
202  *
203  * This interface is fairly simple and it is possible to get the address of the
204  * connection context to do a restore by simply doing a GET call. This avoids the
205  * sanity checks. However, this is not recommended because there may be other side
206  * effects of doing a SET that are necessary when restoring a context and are done
207  * only when the proper option is used to restore the connection context.
208  *
209  *************************************************************************************/
210 typedef struct
211 {
212  uint8_t objVersion;
213  uint16_t objLen;
214  uint8_t **objPtr;
215 } ioctlNVObj_t;

314 /* Network applications may need to remember radio state because the user
315  * application may choose to turn Rx off. These macros help get and restore
316  * the radio Rx state. The macros should be in the same code block at the same level.
317  * The argument 's' is the 'current' radio state and should be set in the code block
318  * with a call to MRFI_GetRadioState() _before_ using the macros.
319  *
320  * Used extensively by NWK but user applications may use them as well. But it is
321  * much more liekly that an application will know the radio state since it likely
322  * will have set it with IOCTL calls. Requires mrfi.h.
323  */
324 #define NWK_CHECK_FOR_SETRX(s) if (MRFI_RADIO_STATE_RX != s) \
325  { \
326  if (MRFI_RADIO_STATE_OFF == s) \
327  { \
328  MRFI_WakeUp(); \
329  } \
330  MRFI_RxOn(); \
331  }
332 
333 #define NWK_CHECK_FOR_RESTORE_STATE(s) if (MRFI_RADIO_STATE_RX != s) \
334  { \
335  if (MRFI_RADIO_STATE_OFF == s) \
336  { \
337  MRFI_Sleep(); \
338  } \
339  else \
340  { \
341  MRFI_RxIdle(); \
342  } \
343  }
344 #endif
